197def bracket_check(maybe_brackets: str | dict):
198    """
199    Check and correct brackets in a kv pair format string\n
200    :param maybe_brackets: The data that may or may not have brackets
201    :type maybe_brackets: `str` | `dict`
202    :return: the corrected string, or a dict
203    """
204    if isinstance(maybe_brackets, dict):
205        pass
206    elif isinstance(maybe_brackets, str):
207        if next(iter(maybe_brackets)) != "{":
208            maybe_brackets = "{" + maybe_brackets
209        if maybe_brackets[-1:] != "}":
210            maybe_brackets += "}"
211    else:
212        raise ValidationError("Check input must be str or dict")
213    return maybe_brackets
